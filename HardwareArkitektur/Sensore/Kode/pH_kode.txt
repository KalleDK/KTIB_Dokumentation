#include <project.h>

#define ph7Default -11
#define ph4Default 84
float ph7;
float ph4;
float phVal;
int16 adcResult;
int16 adcResultmV;

// Omregner til pH-vaerdi. ph4 og ph7 er parametrene som bruges til at kalibrere proben
float mvToPh(float mv, float ph4, float ph7){     
    return ((ph7-mv)/((ph4-ph7)/3)+7);
}

float getPhVal(){
    while(ADC_IsEndConversion(ADC_WAIT_FOR_RESULT)== 0){};
    
    adcResult=ADC_GetResult16(0);
    adcResultmV = ADC_CountsTo_mVolts(0,adcResult);
    
    return mvToPh(adcResultmV,ph4,ph7);
}

CY_ISR(CALIBRATE)
{
    int8 status;
    if(CALI_Read() ==  0){ // vent paa at knappen har vaeret nede i 3 sekunder
    CyDelay(3000);
    if(CALI_Read() == 0)
    status = 1;
    }
    if(status == 1){
    ph7 = getPhVal();
    ph4 = ph4Default + (ph7 - ph7Default); // ph4Default + korrektion 
    RED_LED_Write(0);
    GREEN_LED_Write(1);
    }
}

int main()
{
    ph7 = ph7Default; // Default vaerdi
    ph4 = ph4Default; // Default vaerdi 
    Opamp_1_Start();
    ADC_Start();
    ADC_StartConvert();
    Calibrate_StartEx(CALIBRATE);
    CyGlobalIntEnable; 
    RED_LED_Write(1);
    GREEN_LED_Write(0);
     
   while(1){
    getPhVal();
}
    return 0;
}
